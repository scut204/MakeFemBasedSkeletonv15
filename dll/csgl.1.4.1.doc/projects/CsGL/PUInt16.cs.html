using System;

namespace CsGL.Pointers
{
    /**
     * Inherits from PVoid, uint16 pointer type.
     * @see PVoid
     */
    public unsafe sealed class PUInt16 : PVoid
    {
        /**
         * Constructor/Initializer for n atomic elements.
         * @param n The number of uint16s to allocate upon construction.
         * @see PVoid
         */
        public PUInt16(int n) : base(n)
        {
        }

        /**
         * Returns sizeof(ushort), 2 bytes as per MS specification, this is the 'atomic' element size.
         * @see PVoid
         */
        public override int SizeOfType() { return sizeof(ushort); }
        
        /**
         * Array like accessor, see PVoid.check for exception handling.
         * @param index The index of the uint16 to get.
         * @see PVoid
         */
        public ushort this[int index]
        {
            get 
            { 
                check(index);
                return ((ushort*) data)[index];
            }
            set 
            { 
                check(index);
                ((ushort*) data)[index] = value;
            }
        }
        
        /**
         * Casts a PUInt16 to an unsafe pointer to uint16 (uint16*)
         * @param p The PUInt16 to cast to uint16*
         */
        public static implicit operator ushort*(PUInt16 p) { return (ushort*) p.data; }

        /**
         * Copies a C# uint16 array into a PUInt16 class.
         * @param dst The destination allocated memory. 
         * @param p0 The index to start copying to.
         * @param src The source uint16 array.
         * @param p1 The index to start copying from.
         * @param len The number of atomic (uint16) elements to copy.
         */
        public static void Copy(PUInt16 dst, int p0, ushort[] src, int p1, int len)
        {
            fixed(ushort* psrc = &src[0])
                dst.Copy(dst.data, dst.length, p0, psrc, src.Length, p1, len);
        }

        /**
         * Copies a PUInt16 into a C# array.
         * @param dst The destination C# array.
         * @param p0 The index to start copying to.
         * @param src The allocated memory array to copy from.
         * @param p1 The index to start copying from.
         * @param len The number of atomic (uint16) elements to copy.
         */
        public static void Copy(ushort[] dst, int p0, PUInt16 src, int p1, int len)
        {
            fixed(ushort* pdst = &dst[0])
                src.Copy(pdst, dst.Length, p0, src.data, src.length, p1, len);
        }

        /**
         * See PVoid.
         * @see PVoid
         */
        public static void Copy(PUInt16 dst, int p0, PUInt16 src, int p1, int len)
        {
            dst.Copy(dst.data, dst.length, p0, src.data, src.length, p1, len);
        }
    }
}

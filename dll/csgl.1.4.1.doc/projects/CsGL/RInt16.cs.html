using System;
using System.Runtime.InteropServices;

namespace CsGL.Pointers
{
    /**
     * Simple struct layout for short (int16, 2 bytes).
     */
    [StructLayout(LayoutKind.Sequential)]
    public unsafe struct RInt16
    {
        /**
         * Short pointer.
         */
        short* data;
        
        /**
         * Constructor/Initializer from IntPtr p.
         * @param p Create short pointer from p.
         */
        public RInt16(IntPtr p) { data = (short*)(void*) p; }

        /**
         * Array accessor for memory.
         * @param index Index of the short to access.
         */
        public short this[int index]
        {
            get { return data[index]; }
            set { data[index] = value; }
        }
        
        public static explicit operator IntPtr(RInt16 p) { return (IntPtr) p.data; }
        public static explicit operator RInt16(IntPtr p) { return new RInt16(p); }
        public static RInt16 operator+(RInt16 p, int index)
        {
            RInt16 ret;
            ret.data = p.data + index;
            return ret;
        }
        public static RInt16 operator-(RInt16 p, int index)
        {
            RInt16 ret;
            ret.data = p.data -index;
            return ret;
        }
        public static int operator-(RInt16 a, RInt16 b) { return (int) (a.data -b.data); }
    }
}
